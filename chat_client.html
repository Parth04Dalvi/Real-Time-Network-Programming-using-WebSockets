Backend Setup (Python Server Required)

To enable real-time chat, you must run the accompanying Python server. The client attempts to connect to: http://127.0.0.1:5000

1. Setup Environment:

pip install Flask Flask-SocketIO

2. Server Code (`server.py`):

from flask import Flask, render_template
from flask_socketio import SocketIO, emit, join_room

app = Flask(__name__)
# IMPORTANT: Increase logging level to avoid spamming the console
app.config['SECRET_KEY'] = 'a_secure_secret_key' 
socketio = SocketIO(app, cors_allowed_origins="*")

# In a real app, you would use a user ID, not the username, for security
USERS = {} 
ROOM = "global_chat" 

@socketio.on('connect')
def handle_connect():
    # Client will emit 'join' immediately after this
    pass

@socketio.on('join')
def handle_join(data):
    username = data.get('username', 'Anonymous')
    USERS[request.sid] = username
    join_room(ROOM)
    
    # Notify everyone else that a user joined
    emit('status', {'msg': f'{username} has joined the chat.'}, room=ROOM, skip_sid=request.sid)
    print(f"User {username} joined.")

@socketio.on('message')
def handle_message(data):
    # Ensure data has required fields
    username = data.get('username', 'Unknown')
    message = data.get('message', '')
    
    if message:
        # Broadcast the message to all users in the room
        # We now send the server time for accurate display
        emit('new_message', {'username': username, 'message': message, 'timestamp': time.time() * 1000}, room=ROOM)
        print(f"[{username}]: {message}")

@socketio.on('disconnect')
def handle_disconnect():
    username = USERS.pop(request.sid, 'Anonymous')
    
    # Notify room that the user left
    emit('status', {'msg': f'{username} has left the chat.'}, room=ROOM)
    print(f"User {username} disconnected.")

# Mock endpoint for typing (client would emit this)
@socketio.on('typing')
def handle_typing(data):
    # Broadcast to all others in the room
    emit('typing', {'username': data['username']}, room=ROOM, skip_sid=request.sid)

if __name__ == '__main__':
    import time
    from flask import request
    socketio.run(app, port=5000, debug=False)

3. Run Server:

python server.py
